  cout<< text << " " << text_len << " " <<  pattern << " " << pattern_len << endl;

  int i, j;
  for (i = 0, j = 0; j < pattern_len && i < text_len; i++, j++){
    while (text[i] != pattern[j] && text[i] != '?') {i -= j - 1; j = 0;}
  if (j == pattern_len) cout << i - pattern_len << endl;
  else cout << i << endl;;
  }
